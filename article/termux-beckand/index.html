2 Validasi Data (Mongoose Schema)

Mencegah data kotor masuk ke MongoDB Atlas sangat krusial. Schema Validation mengamankan layer database agar input kosong atau input asal-asalan diblokir seketika.

Schema Lama (Rentang Masalah)
const kontakSchema = new mongoose.Schema({
					  nama: String,
					  pesan: String,
					  waktu: { type: Date, default: Date.now }
					  });
Schema Baru (Aman & Tervalidasi)
const kontakSchema = new mongoose.Schema({
					  nama: {
					  type: String,
					  required: true,
					  trim: true,
					  minlength: 3
					  },
					  pesan: {
					  type: String,
					  required: true,
					  trim: true,
					  maxlength: 500
					  },
					  waktu: { type: Date, default: Date.now }
					  });

Bagaimana kita menangani Error ini di Route Express?

Kita wajib membungkus proses save() dengan skema penanganan try...catch, lalu meneruskan pesan error-nya ke views:

app.post('/tambah', async (req, res) => {
					const { nama, pesan } = req.body;
					try {
					const kontakBaru = new Kontak({ nama, pesan });
					await kontakBaru.save();
					res.redirect('/');
					} catch (error) {
					res.render('index', { 
					kontak: await Kontak.find(), 
					error: "Gagal menambah pesan: Pastikan nama minimal 3 karakter dan pesan tidak kosong!" 
					});
					}
					});

3 Fitur Pencarian Kontak (Query Regex)

Fitur pencarian diimplementasikan menggunakan rute GET. Kita menggunakan pencarian berbasis ekspresi reguler (Regex) dengan opsi 'i' agar tidak sensitif huruf besar/kecil.

GET ROUTE Implementasi Pencarian di server.js: 
app.get('/', async (req, res) => {
					const kataKunci = req.query.search; // menangkap dari input name="search"
					try {
					let query = {};
					if (kataKunci) {
					query.nama = { $regex: kataKunci, $options: 'i' }; 
					}
					const dataKontak = await Kontak.find(query);
					res.render('index', { 
					kontak: dataKontak, 
					searchKunci: kataKunci,
					error: null,
					sudahLogin: req.session.sudahLogin 
					});
					} catch (error) {
					res.status(500).send("Terjadi kesalahan pada server");
					}
					});

5 Gabungan Seluruh Source Code Akhir

Berikut adalah kode lengkap terintegrasi dari seluruh sesi diskusi di atas. Anda bisa langsung menggunakannya sebagai referensi siap pakai.

// Seluruh backend Express + Mongoose + Session 
const express = require('express');
					const mongoose = require('mongoose');
					const bcrypt = require('bcryptjs');
					const session = require('express-session');
					const path = require('path');

					const app = express();

					// Konfigurasi Session
					app.use(session({
					secret: 'kunci_rahasia_aplikasi_kita',
					resave: false,
					saveUninitialized: false
					}));

					// Middleware parser url & json
					app.use(express.urlencoded({ extended: true }));
					app.use(express.json());

					// Set template engine
					app.set('view engine', 'ejs');
					app.set('views', path.join(__dirname, 'views'));

					// MongoDB Connection
					const MONGO_URI = "mongodb+srv://user:pass@cluster.mongodb.net/dbname"; // Ganti dengan URI anda
					mongoose.connect(MONGO_URI)
					.then(() => console.log("🍃 MongoDB Atlas Terkoneksi dengan Sukses"))
					.catch((err) => console.log("Gagal koneksi database: ", err));

					// Database Schemas
					const kontakSchema = new mongoose.Schema({
					nama: { type: String, required: true, trim: true, minlength: 3 },
					pesan: { type: String, required: true, trim: true, maxlength: 500 },
					waktu: { type: Date, default: Date.now }
					});
					const Kontak = mongoose.model('Kontak', kontakSchema);

					const userSchema = new mongoose.Schema({
					username: { type: String, required: true, unique: true },
					password: { type: String, required: true }
					});
					const User = mongoose.model('User', userSchema);

					// --- ROUTES ---

					// 1. READ / TAMPILAN UTAMA (Dengan Fitur Pencarian)
					app.get('/', async (req, res) => {
					const kataKunci = req.query.search;
					try {
					let query = {};
					if (kataKunci) {
					query.nama = { $regex: kataKunci, $options: 'i' };
					}
					const dataKontak = await Kontak.find(query);
					res.render('index', { 
					kontak: dataKontak, 
					searchKunci: kataKunci,
					error: null,
					sudahLogin: req.session.sudahLogin,
					username: req.session.username
					});
					} catch (error) {
					res.status(500).send("Terjadi kesalahan pada server");
					}
					});

					// 2. CREATE / TAMBAH DATA (Dengan Validasi Schema Mongoose)
					app.post('/tambah', async (req, res) => {
					const { nama, pesan } = req.body;
					try {
					const kontakBaru = new Kontak({ nama, pesan });
					await kontakBaru.save();
					res.redirect('/');
					} catch (error) {
					res.render('index', { 
					kontak: await Kontak.find(), 
					searchKunci: '',
					error: "Gagal menambah pesan: Pastikan nama minimal 3 karakter dan pesan tidak kosong!",
					sudahLogin: req.session.sudahLogin,
					username: req.session.username
					});
					}
					});

					// 3. DELETE / HAPUS (Dengan Proteksi Login Sesi)
					app.get('/hapus/:id', async (req, res) => {
					if (!req.session.sudahLogin) {
					return res.status(403).send("Akses ditolak! Anda harus login terlebih dahulu.");
					}
					try {
					await Kontak.findByIdAndDelete(req.params.id);
					res.redirect('/');
					} catch (error) {
					res.status(500).send("Gagal menghapus data");
					}
					});

					// 4. REGISTER ADMIN RAHASIA (Untuk Mendaftar Admin Baru)
					app.post('/register-admin-rahasia', async (req, res) => {
					try {
					const { username, password } = req.body;
					const salt = await bcrypt.genSalt(10);
					const hashedPassword = await bcrypt.hash(password, salt);

					const adminBaru = new User({ username, password: hashedPassword });
					await adminBaru.save();
					res.send("Akun admin berhasil dibuat dengan aman!");
					} catch (error) {
					res.status(500).send("Gagal membuat admin: " + error.message);
					}
					});

					// 5. LOGIN ADMIN
					app.post('/login', async (req, res) => {
					const { username, password } = req.body;
					try {
					const user = await User.findOne({ username });
					if (!user) return res.send("Username atau password salah!");

					const cocok = await bcrypt.compare(password, user.password);
					if (!cocok) return res.send("Username atau password salah!");

					req.session.sudahLogin = true;
					req.session.username = user.username;
					res.redirect('/');
					} catch (error) {
					res.status(500).send("Terjadi kesalahan sistem.");
					}
					});

					// 6. LOGOUT ADMIN
					app.get('/logout', (req, res) => {
					req.session.destroy(() => {
					res.redirect('/');
					});
					});

					app.listen(3000, () => console.log('🚀 Server berjalan lancar di http://localhost:3000'));

					<button type="submit">Cari</button>
					<a href="/"><button type="button">Reset</button></a>
					</form>

					<!-- FORM INPUT DATA BARU -->
					<h3>Tambah Kontak Baru</h3>

					<input type="text" name="nama" placeholder="Nama Lengkap" style="width: 250px;" required><br><br>

					</tr>

					<% if (kontak.length > 0) { %>
					<% kontak.forEach(k => { %>

					<td>
					<% if (typeof sudahLogin !== 'undefined' && sudahLogin) { %>

					<button style="background-color: red; color: white; border: none; padding: 5px 10px; border-radius: 3px;">Hapus</button>
